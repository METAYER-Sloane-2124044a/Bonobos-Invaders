75 unsigned multi::update(MinGL &window){
76  if(cameFromGetPseudo){
77  init(window);

79  }

81  if (wave != lastWave){
82  clearAll(window);

84  //Si pas d'ennemies, c'est qu'il n'y a plus de level donc fin

87  highScore::init(configMapMulti);

89 
90  isWave = false;

92  getPseudo::init(2);
93  if (players[0].score>players[1].score) {
94  return 5;
95  }
96  else {
97  return 6;
98  }
99  }
100  shieldActive = false;
101  isWave= true;
102  }
103  lastWave = wave;
104  //Gestion des entrées claviers
105  if (window.isPressed({'z', false}) && players[0].pos.getY() > players[0].size[1]/2)
106  players[0].pos.setY(players[0].pos.getY()-5);
107  if (window.isPressed({'s', false}) && players[0].pos.getY() < window.getWindowSize().getY()-players[0].size[1]/2)
108  players[0].pos.setY(players[0].pos.getY()+5);
109  if (window.isPressed({'d', false}))
110  players[0].shoot(1,1,torpedos);
111  if (window.isPressed({GLUT_KEY_UP, true}) && players[1].pos.getY() > players[1].size[1]/2)
112  players[1].pos.setY(players[1].pos.getY()-5);
113  if (window.isPressed({GLUT_KEY_DOWN, true}) && players[1].pos.getY() < window.getWindowSize().getY()-players[1].size[1]/2)
114  players[1].pos.setY(players[1].pos.getY()+5);
115  if (window.isPressed({GLUT_KEY_LEFT, true}))
116  players[1].shoot(2,0,torpedos);
117  for (unsigned i(0); i < torpedos.size(); ++i){
118  torpedos[i].update();
119  if (torpedos[i].pos.getX() < 0 || torpedos[i].pos.getX() > window.getWindowSize().getX())
120  torpedos.erase(torpedos.begin()+i);
121  }
122 
123  //Mise à jour des joueurs
124  for (Player &P : players){
125  P.update();
126  if (P.life<=0)
127  ++wave;
128  }
129  std::vector<unsigned> enemiesSup;
130  unsigned cpt=0;
131  for (Enemy &E : enemies){
132  E.update(enemies, torpedos,window); //on met a jour les invaders
133  if (E.pos.getX()<players[0].size[0]*2) {
134  enemiesSup.push_back(cpt); //on compte les invaders qui ont dépassé le camp
135  if (E.alive)
136  --players[0].life;
137  }
138  else if (E.pos.getX()>window.getWindowSize().getX()-players[1].size[0]*2) {
139  enemiesSup.push_back(cpt);
140  if (E.alive)
141  --players[1].life;
142  }
143  ++cpt;
144  }
145  for (unsigned i:enemiesSup) //on suprime les invaders qui ont dépassé le camp allié
146  enemies.erase(enemies.begin() + i);
147 
148  //Gestion des collisions
149  std::vector<std::vector<char>> returnVec;
150  collision::genColl(torpedos, players);
151  returnVec = collision::collision(players,torpedos,enemies,shields,startgame);
152 
153  for (std::vector<char> collide : returnVec){
154  switch (collide[0]) {
155  case 't':
156  switch (collide[2]) {
157  case 'p':
158  if (collide[4] == 1){
159  //torpille à touché joueur 1
160  --players[1].life;
161  if (collide[1] != 0) score::scorePlayer(players[0]);
162  }
163  else{
164  //torpille à touché joueur 2
165  --players[0].life;
166  if (collide[1] != 0) score::scorePlayer(players[1]);
167  }
168  break;
169  case 'e':
170  //torpille à touché un enemie
171  score::scoreInvader(collide[1],enemies[collide[3]],players);
172  enemies[collide[3]].alive = false;
173  //Activation du bouclier si nessecaire
174  if (not shieldActive){
175  if (not chekEnemiesInLive(enemies,'g')) {
176  for (unsigned i(0); i < shields.size() / 2; ++i) {
177  shields[i].life = 3;
178  shieldActive = true;
179  }
180  }
181  if (not chekEnemiesInLive(enemies,'d')) {
182  for (unsigned i(shields.size()/2); i < shields.size(); ++i) {
183  shields[i].life = 3;
184  shieldActive = true;
185  }
186  }
187  }
188  break;
189  }
190  break;
191  //collision entre joueur et enemies
192  case 'p':
193  enemies[collide[2]].alive = false;
194  break;
195  //collision entre une torpille et le start
196  case 's':
197  startgame.clear();
198  wave = 1;
199  isWave=true;
200  break;
201  }
202  }
203 
204  if (isWave && not chekEnemiesInLive(enemies,'g') && not chekEnemiesInLive(enemies, 'd')){ //Mort subite quand y a plus d'enemies
205  for(Player &P : players){
206  P.life = 1;
207  P.pos.setY(window.getWindowSize().getY()/2);
208  }
209  torpedos.clear();
210  isWave= false;
211  }
212 
213  return 1;
